~~~ sql
-- 当前日期
SELECT GETDATE()
-- 目前第几天
SELECT DAY(GETDATE())
-- 上个月最后一天
SELECT GETDATE() - DAY(GETDATE())
-- 加上一个月
SELECT DATEADD(MONTH, 1, GETDATE()) - DAY(GETDATE())

-- 获得当月天数
SELECT DAY(DATEADD(MONTH, 1, GETDATE()) - DAY(GETDATE()))
-- 获得上一个月天数
SELECT DAY(GETDATE() - DAY(GETDATE()))
-- 获得下一个月天数
SELECT DAY(DATEADD(MONTH, 2, GETDATE()) - DAY(GETDATE()))
-- 当前日期是星期的第几天
SELECT DATEPART(WEEKDAY, GETDATE())

-- 当前月第一天
SELECT DATEADD(DAY, 1 - DAY(GETDATE()), GETDATE())

-- 当前月最后一天
SELECT DATEADD(MONTH, 1, DATEADD(DAY, 0 - DAY(GETDATE()), GETDATE()))


-- 周一
SELECT CONVERT(VARCHAR(10), DATEADD(DAY, -DATEPART(WEEKDAY, GETDATE()) + 2, GETDATE()), 120)
-- 周六
SELECT CONVERT(VARCHAR(10), DATEADD(DAY, -DATEPART(WEEKDAY, GETDATE()) + 7, GETDATE()), 120)

SELECT -- 上个月天数
       DAY(GETDATE() - DAY(GETDATE())) AS LastMonthDays,
       -- 本月天数
       DAY(DATEADD(mm, 1, GETDATE()) - DAY(GETDATE())) AS CurrentMonthDays,
       -- 本月第一天是周几
       DATEPART(WEEKDAY, GETDATE() - DAY(GETDATE()) + 1) AS FirstWeekDay

-- 年
SELECT DATEPART(YEAR, GETDATE()), DATENAME(YEAR, GETDATE()), YEAR(GETDATE())
-- 月
SELECT DATEPART(MONTH, GETDATE()), DATENAME(MONTH, GETDATE()), MONTH(GETDATE())
-- 日
SELECT DATEPART(DAY, GETDATE()), DATENAME(DAY, GETDATE()), DAY(GETDATE())
-- 时
SELECT DATEPART(HOUR, GETDATE()), DATENAME(HOUR, GETDATE())
-- 分
SELECT DATEPART(MINUTE, GETDATE()), DATENAME(MINUTE, GETDATE())
-- 秒
SELECT DATEPART(SECOND, GETDATE()), DATENAME(SECOND, GETDATE())
-- 微秒
SELECT DATEPART(MS, GETDATE())
-- 毫秒
SELECT DATEPART(MCS, GETDATE())
-- 纳秒
SELECT DATEPART(NS, GETDATE())
-- 季度
SELECT DATEPART(QQ, GETDATE())
-- 一年中的第几周
SELECT DATEPART(WEEK, GETDATE())
-- 周几
SELECT DATEPART(WEEKDAY, GETDATE()) - 1
~~~
